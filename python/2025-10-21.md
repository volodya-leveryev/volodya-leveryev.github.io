# Перечисления

```python
from enum import Enum, StrEnum, Flag, auto

# Enum - базовый класс для создания перечислений
# IntEnum - базовый класс для создания перечисления с помощью последовательных целых чисел
# StrEnum - базовый класс для создания перечисления с помощью строк
# Flag - базовый класс для создания перечисления с помощью битовох флагов - степеней двойки

class WeekdayVariant1(Enum):
    MONDAY = 1
    TUESDAY = 2
    WEDNESDAY = 3
    THURSDAY = 4
    FRIDAY = 5
    SATURDAY = 6
    SUNDAY = 7

    @classmethod
    def from_date(cls, date):
        return cls(date.isoweekday())

class Color(StrEnum):
    RED = 'r'
    GREEN = 'g'
    BLUE = 'b'
    UNKNOWN = auto()

class WeekdayVariant2(Flag):
    MONDAY = auto()
    TUESDAY = auto()
    WEDNESDAY = auto()
    THURSDAY = auto()
    FRIDAY = auto()
    SATURDAY = auto()
    SUNDAY = auto()

    # Можно создавать комбинированные значения
    WEEKEND = SATURDAY | SUNDAY
```
